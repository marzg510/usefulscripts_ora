# Storage

```SQL
select
  FILE#,
  ts.name TABLESPACE_NAME,
  df.name DATAFILE_NAME,
  bytes/1024/1024 MBYTES,
  status DATAFILE_STATUS,
  to_char(checkpoint_time,'YYYY-MM-DD HH:MI.SS') CHECKPOINT_TIME
from 
  v$tablespace ts, 
  v$datafile df
where ts.ts#=df.ts#
order by file#
/

```

```SQL
select 
OWNER,
SEGMENT_NAME,
PARTITION_NAME,
SEGMENT_TYPE,
TABLESPACE_NAME,
HEADER_FILE,
HEADER_BLOCK,
BYTES,
BYTES / 1024 / 1024 / 1024 as BYTES_GIB,
BLOCKS,
EXTENTS,
INITIAL_EXTENT,
NEXT_EXTENT,
MIN_EXTENTS,
MAX_EXTENTS,
PCT_INCREASE,
FREELISTS,
FREELIST_GROUPS,
RELATIVE_FNO,
BUFFER_POOL
from DBA_SEGMENTS
where OWNER != 'SYS'
order by bytes desc
/
```
